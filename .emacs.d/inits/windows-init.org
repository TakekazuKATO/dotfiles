#+begin_src emacs-lisp
(defun w32-browser 
  (doc) 
  (w32-shell-execute 1 doc))

(eval-after-load "dired" 
  '(define-key dired-mode-map [f6] 
     (lambda () 
       (interactive) 
       (w32-browser (dired-replace-in-string "/" "\\" (dired-get-filename))))))
#+end_src
